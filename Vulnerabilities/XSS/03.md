# =========================================================
# URL COLLECTION
# =========================================================

```bash
# Single Target
echo "www.post.ch" | waybackurls | tee urls.txt
gau www.post.ch | anew urls.txt

katana -u https://www.post.ch \
  -jc \
  -kf all \
  -d 5 \
  -aff \
  -fx \
  -xhr-extraction \
  -js-crawl \
  -known-files all \
  -automatic-form-fill \
  -silent | anew urls.txt


# Multi Target
cat hosts | waybackurls | sort -u > urls.txt

katana -list hosts \
  -d 3 \
  -jc \
  -kf all \
  -aff \
  -fx \
  -xhr-extraction \
  -js-crawl \
  -silent | anew urls.txt
```


# =========================================================
# FILTERING
# =========================================================

```bash
# JS files
grep -Ei '\.js(\?|$)' urls.txt | sort -u > js_urls.txt

# URLs with params
grep '=' urls.txt | sort -u > params.txt

# Potential API endpoints
grep -Ei '/api/|/v1/|/v2/|graphql|auth|oauth|saml|account|profile|session' \
  urls.txt | sort -u > api_candidates.txt

# Interesting extensions
grep -Ei '\.(json|map|xml|txt|bak|old|zip)(\?|$)' \
  urls.txt | sort -u > misc_files.txt
```


# =========================================================
# PARAM TESTING
# =========================================================

```bash
# XSS candidate
cat params.txt | gf xss

# Redirect candidate
cat params.txt | gf redirect

# SSRF/OpenURL candidate
cat params.txt | gf ssrf

# SSTI candidate
cat params.txt | gf ssti

# LFI candidate
cat params.txt | gf lfi


# Reflection quick check
cat params.txt | \
qsreplace 'xsszxczx' | \
httpx -silent -mc 200 | \
grep -F 'xsszxczx'


# HTML reflection only
cat params.txt | \
qsreplace '\"><xsszxczx>' | \
httpx -silent -mc 200 | \
grep -F '<xsszxczx>'
```

# =========================================================
# DOWNLOAD JS
# =========================================================

```bash
mkdir -p js

cat js_urls.txt | xargs -P8 -I{} sh -c '
url="{}"
filename=$(echo "$url" | md5sum | cut -d" " -f1).js

curl -ksL \
  --max-time 20 \
  --retry 2 \
  --retry-delay 1 \
  "$url" \
  -o "js/$filename"
'
```

# =========================================================
# HIGH SIGNAL SOURCE + SINK
# =========================================================

```bash
for f in $(find js -name "*.js"); do
  grep -qE \
  'location\.|URLSearchParams|hash|postMessage|localStorage|sessionStorage' "$f" && \
  grep -qE \
  'innerHTML|outerHTML|document\.write|insertAdjacentHTML|eval\(|new Function' "$f" && \
  echo "[POTENTIAL DOM XSS] $f"
done
```
